// ============================================================
// IMPULSE OBJECT ENGINE — PRIMARY TF
// Основа: ta.pivothigh / ta.pivotlow с задержкой pivotLen баров
// Три freeze-условия из донора: higher-low + ATR gate + sequence counter
// raw impulse candidate → не owner, не POI, не confirmed
// ============================================================

int   pivotLen = 5

// Pivot detection на PRIMARY TF
float primPivHigh = request.security(syminfo.tickerid, primaryTf, ta.pivothigh(high, pivotLen, pivotLen), lookahead=barmerge.lookahead_off)
float primPivLow  = request.security(syminfo.tickerid, primaryTf, ta.pivotlow(low,  pivotLen, pivotLen), lookahead=barmerge.lookahead_off)

bool primNewPivHigh = not na(primPivHigh)
bool primNewPivLow  = not na(primPivLow)

// Impulse leg candidate — PRIMARY TF
var float primLegBasePrice = na
var float primLegTermPrice = na
var int   primLegDir       = 0    // 1=bull -1=bear
var bool  primLegExists    = false

// Bull leg: base=pivotLow → terminal=pivotHigh
if primNewPivLow
    primLegBasePrice := primPivLow
    primLegDir       := 1
    primLegExists    := true

// Bear leg: base=pivotHigh → terminal=pivotLow
if primNewPivHigh
    primLegBasePrice := primPivHigh
    primLegDir       := -1
    primLegExists    := true

// Terminal — последний экстремум в направлении ноги
var float primLegTerminal = na
if primLegDir == 1 and primNewPivHigh
    primLegTerminal := primPivHigh
if primLegDir == -1 and primNewPivLow
    primLegTerminal := primPivLow

// Leg range
float primLegRange = primLegExists and not na(primLegTerminal) ? math.abs(primLegTerminal - primLegBasePrice) : na

// ATR gate — шумозащита (из донора)
float primAtr = request.security(syminfo.tickerid, primaryTf, ta.atr(14), lookahead=barmerge.lookahead_off)
bool  primLegAboveAtr = not na(primLegRange) and primLegRange > primAtr

// Retrace >= 0.5 — confirmation rule
var float primRetrace = na
if primLegExists and not na(primLegTerminal) and not na(primLegBasePrice)
    primRetrace := primLegDir == 1 ? (primLegTerminal - close) / primLegRange :
                                     (close - primLegTerminal) / primLegRange

bool primLegConfirmed = not na(primRetrace) and primRetrace >= 0.5 and primLegAboveAtr
